  <script>
    document.addEventListener('DOMContentLoaded', function() {
      // Update last updated time with current date
      function updateLastUpdated() {
        const now = new Date();
        const options = { 
          year: 'numeric', 
          month: 'long', 
          day: 'numeric', 
          hour: '2-digit', 
          minute: '2-digit',
          timeZoneName: 'short'
        };
        const formattedDate = now.toLocaleDateString('en-US', options);
        document.getElementById('lastUpdated').textContent = `Last updated: ${formattedDate}`;
      }
      
      // Update immediately and then every minute
      updateLastUpdated();
      setInterval(updateLastUpdated, 60000);
      
      // Search functionality
      const searchInput = document.getElementById('gameSearch');
      const filterButton = document.getElementById('filterButton');
      const gameCards = document.querySelectorAll('.game-card');
      
      searchInput.addEventListener('input', function() {
        const searchTerm = this.value.toLowerCase();
        
        gameCards.forEach(card => {
          const gameName = card.querySelector('.game-name').textContent.toLowerCase();
          const gameStatus = card.querySelector('.status-indicator').textContent.toLowerCase();
          const gameNotes = card.querySelector('.detail-value:last-child').textContent.toLowerCase();
          
          if (gameName.includes(searchTerm) || gameStatus.includes(searchTerm) || gameNotes.includes(searchTerm)) {
            card.classList.remove('hidden');
          } else {
            card.classList.add('hidden');
          }
        });
      });
      
      // Filter functionality
      let activeFilter = 'all';
      
      filterButton.addEventListener('click', function() {
        switch(activeFilter) {
          case 'all':
            activeFilter = 'good';
            filterByStatus('good');
            this.textContent = 'Filter: Good';
            break;
          case 'good':
            activeFilter = 'unsupported';
            filterByStatus('unsupported');
            this.textContent = 'Filter: Unsupported';
            break;
          case 'unsupported':
            activeFilter = 'detected';
            filterByStatus('detected');
            this.textContent = 'Filter: Detected';
            break;
          case 'detected':
            activeFilter = 'unsure';
            filterByStatus('unsure');
            this.textContent = 'Filter: Unsure';
            break;
          case 'unsure':
            activeFilter = 'all';
            filterByStatus('all');
            this.textContent = 'Filters';
            break;
        }
      });
      
      function filterByStatus(status) {
        gameCards.forEach(card => {
          if (status === 'all') {
            card.classList.remove('hidden');
          } else {
            if (card.classList.contains(status)) {
              card.classList.remove('hidden');
            } else {
              card.classList.add('hidden');
            }
          }
        });
      }
      
      // Animation delay for cards
      gameCards.forEach((card, index) => {
        card.style.animationDelay = (1.9 + index * 0.05) + 's';
      });
      
      // Remove intro overlay after animations
      setTimeout(() => {
        const overlay = document.querySelector('.intro-overlay');
        if (overlay) {
          overlay.style.display = 'none';
        }
      }, 3000);
    });
  </script>
